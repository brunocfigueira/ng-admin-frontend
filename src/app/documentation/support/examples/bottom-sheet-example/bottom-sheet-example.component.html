          MyDisplayContent.component.ts
          ------------------------------------------------------------------------
          export class MyDisplayContentComponent {
          private _bottomSheetRef =
              inject<MatBottomSheetRef<MyDisplayContentComponent>>(MatBottomSheetRef);

              constructor(@Inject(MAT_BOTTOM_SHEET_DATA) public arrayData: any[]) {
              }

              openLink(event: MouseEvent): void  {
                  this._bottomSheetRef.dismiss();
                  event.preventDefault();
               }
          }
          
      
        
          //using the service to display the content

          import BottomSheetService from '../../../../services/common/bottom-sheet.service';

          export class YourComponent implements OnInit &lbrace
            constructor(private bottomSheetService: BottomSheetService) {}
            ngOnInit(): void {
              const arrayData = [{
                text: 'Some Text',
                link: '/internalLink'
              }]
              this.bottomSheetService.open(MyDisplayContentComponent, arrayData)
            }
          }
