Use this data to define the height and width of a component based on the browser screen reference.

Explore all possible component attributes.

        
          import WindowSizeService from '../../window-size.service';

          export class YourComponent implements OnInit &lbrace
            constructor(private windowSizeService: WindowSizeService) {}
            ngOnInit(): void {


              const dataWindow =  this.windowSizeService.getCurrentSize();
              // screen width
              console.log(dataWindow.width);
              // screen height
              console.log(dataWindow.height);

              // percentage 30% for width
              const widthPercentage =  this.windowSizeService.getCurrentWidthByPercentage(0.3);
              // screen width
              console.log(widthPercentage);
              // percentage 30% for height
              const heightPercentage =  this.windowSizeService.getCurrentHeightByPercentage(0.3);
              // screen height
              console.log(heightPercentage);
            }
          }
