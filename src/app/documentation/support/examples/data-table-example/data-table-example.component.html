Use the DataTableComponent in a simple way to display tabulated data.

Explore all possible component attributes.

         
           import DataTableComponent from '../../shared/common/data-table/data-table.component';
           export class YourComponent implements {
           dataColumns: DataTableColumn[] = [
               {key: 'id', title: 'Id'},
               {key: 'name', title: 'Name'},
            ];
            dataTable: any[] = [
                {id: 1, name: 'Name One'},
                {id: 2, name: 'Name Two'},
                ...
            ];
           }
         
       
         
           your.component.html

           <app-data-table
                    [maxHeight]="500"
                    [isHttpRequest]="false"
                    [dataColumns]="dataColumns"
                    [dataTable]="dataTable"
            ></app-data-table>

Use the Data Table component in an advanced way to display tabulated data.

Explore all possible component attributes.

         
           // all attributes
           your.component.html

            <app-data-table
                [maxHeight]="500"
                [isHttpRequest]="false"
                [dataColumns]="dataColumns"
                [dataTable]="dataTable"
                [showActionExportPdf]="true"
                (onExportPdf)="eventExportPdf($event)"
                [showActionExportCsv]="true"
                (onExportCsv)="eventExportCsv($event)"
                [showActionRefresh]="true"
                (onRefresh)="eventRefresh()"
                [showActionCreate]="true"
                (onCreate)="eventCreateRecord()"
                [showInputSearch]="true"
                (onSearch)="eventSearchRecords($event)"
                [showActionEdit]="true"
                (onEdit)="eventEditRecord($event)"
                [showActionDelete]="true"
                (onDelete)="eventDeleteRecord($event)"
                [showActionDeleteAll]="true"
                (onDeleteAll)="eventDeleteAllRecords($event)"
                [showPagination]="true"
                (onPageChange)="eventPageChange($event)"
                [showActionDetails]="true"
                [maxHeightItemDetails]="250"
                [showInputSearchItemDetails]="true"
                [showPaginationItemDetails]="true"
                [itemDetailsColumns]="itemDetailsColumns"
                [itemDetailsTable]="itemDetailsTable"
                (onViewItemDetails)="eventViewItemDetails($event)"
            ></app-data-table>
